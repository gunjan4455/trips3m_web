export const phoneNumber = (value) => {
  if (!value) {
    return undefined;
  }
  if ((`${value}` === `${parseInt(value, 10)}`) &&
    value.length >= 5 &&
    value.length <= 15
  ) {
    return undefined;
  }
  return 'Phone number not valid';
};
