import React from 'react';
import './Sidenav.scss';
import PropTypes from 'prop-types';
import { ProfileIcon, HomeIcon, VactionIcon, HoneymoonPackages, FamilyIcon,
  WorkIcon, OffersIcon, IndianIconBlack, InternationalIconBlack,
  BlogIcon, TestimonailsIcon, FAQIcon, AllTrips, ContactIcon,
  LocationMarkerIcon, Seasonal, } from 'helpers/Icon/Icon';
import { Link } from 'react-router';
import { trackSegment, DESTINATION_LIST_CLICK, DFEAULT_VALUE, PACKAGE_LIST_CLICK } from '../../actions/segmentEvents';
import HotelLinks from './HotelLinks';
 
const SidenavMenu = ({ selectHandler, isAuthenticated, handleLogout, trackWeekendListClicked }) => {
  const section = 'Global Menu';
  const destinationListClick = (type, cta) => {
    selectHandler();
    let destinationName;
    if (type === 'weekend') {
      destinationName = 'Delhi';
      trackWeekendListClicked({
        section,
        object: '',
        cta,
        destination: destinationName
      });
    } else {
      trackSegment({
        event: DESTINATION_LIST_CLICK,
        section: section,
        object: '',
        cta: cta,
        category: `Destination List: ${type}`,
        destination_list: type,
        destination_id: type,
        value: DFEAULT_VALUE,
        label: `Destination Type: ${type}`
      });
    }
    return true;
  };
 
  const packageListClick = (category, cta) => {
    selectHandler();
    trackSegment({
      event: PACKAGE_LIST_CLICK,
      section: section,
      category: 'Destination:',
      cta: cta,
      object: 'Destination/0',
      label: `Package Type: ${category}`,
      destination_list: '',
      destination_id: 0,
      value: DFEAULT_VALUE
    });
  };
 
  return (
    <div className="row row-">
      <ul className="list-side-menu mb24 pb15">
        {isAuthenticated ?
          <div>
            <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
              <meta itemProp="name" content="Logout" />
              <Link itemProp="url" onClick={handleLogout} className="ripple"><div className="wave-dark" />
                <span className="sidenav-icons"><ProfileIcon /></span>
                Logout
              </Link>
            </li>
            <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement" className="alltrips">
              <meta itemProp="name" content="All Trips" />
              <a itemProp="url" href="/travellers/dashboard" onClick={selectHandler} className="ripple">
                <div className="wave-dark" />
                <span className="sidenav-icons">
                      <AllTrips />
                    </span>
                All Trips
              </a>
            </li>
          </div> :
          <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
            <meta itemProp="name" content="Login/Sign up" />
            <Link itemProp="url" to="/users/sign_in" className="at_signUp ripple"><div className="wave-dark" />
              <span className="sidenav-icons"><ProfileIcon /></span>
              Login/Sign up
            </Link>
          </li>
        }
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="Home" />
          <Link itemProp="url" to="/" onClick={selectHandler} className="at_homeLink ripple"><div className="wave-dark" /><span className="sidenav-icons"><HomeIcon /></span>Home
          </Link>
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="Tour Packages" />
          <Link itemProp="url" to="/tour-packages" onClick={() => packageListClick('Tour', 'Tour Packages')} className="ripple"><div className="wave-dark" /><span className="sidenav-icons"><VactionIcon /></span>Tour Packages</Link>
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="Honeymoon Packages" />
          <Link itemProp="url" to="/honeymoon-packages" onClick={() => packageListClick('Honeymoon', 'Honeymoon Packages')} className="ripple"><div className="wave-dark" /><span className="sidenav-icons"><HoneymoonPackages /></span>Honeymoon Packages</Link>
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="Family Packages" />
          <Link itemProp="url" to="/family-packages" onClick={() => packageListClick('Family', 'Family Packages')} className="ripple"><div className="wave-dark" /><span className="sidenav-icons"><FamilyIcon /></span>Family Packages</Link>
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="Indian Destinations" />
          <Link itemProp="url" to="/All-Places/?is_international=false" className="ripple"onClick={() => destinationListClick('Indian', 'Indian Destinations')}><div className="wave-dark" /><span className="sidenav-icons"><IndianIconBlack /></span>Indian Destinations</Link>
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="International Destinations" />
          <Link itemProp="url" to="/All-Places?is_international=true" className="ripple"onClick={() => destinationListClick('International', 'International Destinations')}><div className="wave-dark" /><span className="sidenav-icons"><InternationalIconBlack /></span>International Destinations</Link>
        </li>
        <li>
          <HotelLinks />
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="Destination Guide" />
          <Link itemProp="url" to="/india-tourism" className="ripple"><div className="wave-dark" /><span className="sidenav-icons"><LocationMarkerIcon /></span>Destination Guide</Link>
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="Seasonal Packages" />
          <Link itemProp="url" to="/seasonal-packages" className="ripple"><div className="wave-dark" /><span className="sidenav-icons"><Seasonal /></span>Seasonal Packages</Link>
        </li>
      </ul>
      <ul className="list-side-menu m0 pb15">
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="How It Works" />
          <a itemProp="url" href="https://traveltriangle.com/how_it_works" className="ripple"><div className="wave-dark" /><span className="sidenav-icons"><WorkIcon /></span>How It Works</a>
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="Offers" />
          <a itemProp="url" href="https://traveltriangle.com/offers" className="ripple"><div className="wave-dark" /><span className="sidenav-icons"><OffersIcon /></span>Offers</a>
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="Blog" />
          <a itemProp="url" href="https://traveltriangle.com/blog" className="ripple"><div className="wave-dark" /><span className="sidenav-icons"><BlogIcon /></span>Blog</a>
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="Testimonials" />
          <a itemProp="url" href="https://traveltriangle.com/testimonials" className="ripple"><div className="wave-dark" /><span className="sidenav-icons"><TestimonailsIcon /></span>Testimonials</a>
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="FAQ" />
          <a itemProp="url" href="https://traveltriangle.com/FAQs" className="ripple"><div className="wave-dark" /><span className="sidenav-icons"><FAQIcon /></span>FAQ</a>
        </li>
        <li itemScope="itemscope" itemType="https://schema.org/SiteNavigationElement">
          <meta itemProp="name" content="Contact Us" />
          <a itemProp="url" href="https://traveltriangle.com/contact_us" className="ripple"><div className="wave-dark" /><span className="sidenav-icons"><ContactIcon /></span>Contact Us</a>
        </li>
      </ul>
      <div className="p15">
        <p className="f14 mb8">For any help</p>
        <h6 className="f16 m0 fw9"><a href="tel:18001235555">1800-123-5555</a></h6>
      </div>
    </div>
  );
};
 
SidenavMenu.propTypes = {
  selectHandler: PropTypes.func,
  isAuthenticated: PropTypes.bool,
  handleLogout: PropTypes.func.isRequired,
  trackWeekendListClicked: PropTypes.func.isRequired,
};
 
SidenavMenu.defaultProps = {
  selectHandler: () => {},
  isAuthenticated: false
};
export default SidenavMenu;
