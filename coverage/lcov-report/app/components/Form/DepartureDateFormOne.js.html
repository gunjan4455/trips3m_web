function currentWeek(date) {
  const date_new = new Date(date);
  const thisDay = date_new.getDate();
  const newDate = date_new;
  newDate.setDate(1);
  const digit = newDate.getDay();
  const Q = (thisDay + digit) / 7;
  const R = (thisDay + digit) % 7;
  if (R !== 0)
    return Math.ceil(Q);
  else
    return Q;
}
