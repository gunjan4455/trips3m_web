export const isApiError = (code) => {
  const codes = [
    PERMANENT_REDIRECT, TEMPORARY_REDIRECT, BAD_REQUEST, UNAUTHORIZED, PAYMENT_REQUIRED, FORBIDDEN, NOT_FOUND,
    METHOD_NOT_ALLOWED, NOT_ACCEPTABLE, PROXY_AUTHENTICATION_REQUIRED, REQUEST_TIMEOUT, CONFLICT, GONE,
    LENGTH_REQUIRED, PRECONDITION_FAILED, REQUEST_ENTITY_TOO_LARGE, REQUEST_URI_TOO_LONG, UNSUPPORTED_MEDIA_TYPE,
    REQUESTED_RANGE_NOT_SATISFIABLE, EXPECTATION_FAILED, I_AM_A_TEAPOT, UNPROCESSABLE_ENTITY, LOCKED,
    FAILED_DEPENDENCY, UPGRADE_REQUIRED, CONNECTION_TIMEOUT, INTERNAL_SERVER_ERROR, NOT_IMPLEMENTED, BAD_GATEWAY,
    SERVICE_UNAVAILABLE, GATEWAY_TIMEOUT, HTTP_VERSION_NOT_SUPPORTED, VARIANT_ALSO_NEGOTIATES, INSUFFICIENT_STORAGE,
    NOT_EXTENDED
  ];
 
  if (code) {
    return codes.indexOf(code) !== -1;
  }
  return false;
};
