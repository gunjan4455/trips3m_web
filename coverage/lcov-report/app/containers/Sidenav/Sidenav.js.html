import { connect } from 'react-redux';
import { push } from 'react-router-redux';
import {
  isLoggedIn as isAuthenticated,
  user,
  shouldDisplayLatestTrip
} from 'reducers/auth';
import {
  trackWeekendListClicked
} from 'actions/segmentEvents';
import { logout, load as loadUserDetails } from 'actions/auth';
import Sidenav from 'components/Sidenav/Sidenav';
 
const mapStateToProps = state => ({
  isAuthenticated: isAuthenticated(state),
  userDetails: user(state),
  shouldDisplayLatestTrip: shouldDisplayLatestTrip(state),
  pathName: (state.routing && state.routing.locationBeforeTransitions.pathname) || '',
});
 
const mapDispatchToProps = dispatch => ({
  handleLogout: () => dispatch(logout()),
  replaceState: params => dispatch(push(params)),
  trackWeekendListClicked: params => dispatch(trackWeekendListClicked(params)),
  loadUserDetails: params => dispatch(loadUserDetails(params)),
});
 
export default connect(mapStateToProps, mapDispatchToProps)(Sidenav);
