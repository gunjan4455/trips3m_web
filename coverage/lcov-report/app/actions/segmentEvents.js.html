    scroll25 = true;
    trackSegmentEvent('scroll25');

    if (!scroll25) {
      trackSegmentEvent('scroll25');
      scroll25 = true;
    }
    trackSegmentEvent('scroll50');
    scroll50 = true;

      trackSegmentEvent('scroll25');
      scroll25 = true;
    }
    if (!scroll50) {
      trackSegmentEvent('scroll50');
      scroll50 = true;
    }

    scroll75 = true;
