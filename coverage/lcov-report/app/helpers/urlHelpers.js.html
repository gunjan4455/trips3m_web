const getAppSchemeUrl = (location, pageNumber, os) => {
  let baseUrl = `traveltriangle.com${location.pathname}`;
 
  if (os === 'ios') {
    baseUrl = location.pathname.substring(1);
  }
 
  if (pageNumber !== 1) {
    baseUrl = `${baseUrl}/page/${pageNumber}`;
  }
 
  if (location.search) {
    baseUrl = `${baseUrl}/${location.search}`;
  }
 
  return baseUrl;
};
