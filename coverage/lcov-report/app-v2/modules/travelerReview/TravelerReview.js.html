import React from 'react';
import PropTypes from 'prop-types';
import { Link } from 'react-router';
 
import TravelerReviewCard from './TravelerReviewCard';
import Swiper from 'components/Swiper/Swiper';
import Heading from '../shared/Heading';
 
const TravelerReview = ({
                          testimonials = [],
                          isSeasonalCategory = false,
                          uniqueKey,
                          showViewAll,
                          data
                        }) => {
  if (!testimonials.length) {
    return null;
  }
 
  const testimonialsLink = (testimonials.destination !== undefined) ?
    `/testimonials?destination=${testimonials.destination}` : '/testimonials';
 
  return (
    <div className="slider-overflow-buttons sbcw">
      <div className="traveler-review">
        <div className="relative">
          <Heading data={data} />
          {
            showViewAll ?
              <Link target={'_blank'} to={testimonialsLink}
                    className="iblock f14 fw7 r0 t0 absolute at_readmorebtn at_testimonial_viewall">
                View All
              </Link> : null
          }
        </div>
        <div className="clearfix">
          <Swiper key={uniqueKey} slideClass="slide" spaceBetween={15} slidesPerView="auto"
                  ssrAnimation={{ width: 314, height: 257 }} rebuildOnUpdate>
            {
              testimonials
                .map((testimonial, i) =>
                  <div key={`listing_traveler_review_${testimonial.id}_${i}`}>
                    <TravelerReviewCard
                      testimonial={testimonial}
                      isSeasonalCategory={isSeasonalCategory}
                    />
                  </div>
                )
            }
          </Swiper>
        </div>
      </div>
    </div>
  );
};
 
TravelerReview.propTypes = {
  testimonials: PropTypes.array,
  uniqueKey: PropTypes.string,
  isSeasonalCategory: PropTypes.bool,
  showViewAll: PropTypes.bool,
  data: PropTypes.object
};
 
TravelerReview.defaultProps = {
  testimonials: [],
  uniqueKey: 'traveler-review',
  isSeasonalCategory: false,
  showViewAll: true,
  data: {
    title: "Traveler Reviews",
    heading: "Read on to find out why our customers love us!"
  }
};
 
export default TravelerReview;
