import {
  PREFERENCE_OPTIONS_LOAD,
  PREFERENCE_OPTIONS_LOAD_FAIL,
  PREFERENCE_OPTIONS_LOAD_SUCCESS,
  UPDATE_RTRIP_LOAD,
  UPDATE_RTRIP_LOAD_SUCCESS,
  UPDATE_RTRIP_LOAD_FAIL
} from './action';
 
const defaultPreferenceOptionsState = {
  loaded: false,
  loading: false,
  updated: false,
  updating: false,
  preferenceOptions: {},
  errors: []
};
 
export default function preferencesSection(state = defaultPreferenceOptionsState, action) {
  switch (action.type) {
    /**
     * Loading new preference hence initializing with default state
     */
    case PREFERENCE_OPTIONS_LOAD: {
      return {
        ...state,
        loaded: false,
        loading: true
      };
    }
    case PREFERENCE_OPTIONS_LOAD_SUCCESS: {
      const preferenceOptions = action.result.data.requestedTrip;
      return {
        ...state,
        preferenceOptions,
        loaded: true,
        loading: false
      };
    }
    case PREFERENCE_OPTIONS_LOAD_FAIL: {
      return {
        ...state,
        loaded: false,
        loading: false
      };
    }
    case UPDATE_RTRIP_LOAD: {
      return {
        ...state,
        updated: false,
        updating: false
      };
    }
    case UPDATE_RTRIP_LOAD_SUCCESS: {
      const preferenceOptions = action.result.data.requestedTrip;
      return {
        ...state,
        preferenceOptions: { ...state.preferenceOptions, ...preferenceOptions },
        updated: true,
        updating: false
      };
    }
    case UPDATE_RTRIP_LOAD_FAIL: {
      console.log('action...', action);
      console.log('action.result...', action.result);
      return {
        ...state,
        updated: false,
        updating: false,
        errors: action.error.errors
      };
    }
    default: {
      return state;
    }
  }
}
