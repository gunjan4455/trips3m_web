const getLabel = (value) => {
  let label = '';
  if (!value) {
    label = 'Not Required';
  }
  else if (value.toLowerCase() === 'included') {
    label = 'Yes';
  }
  else if (value.toLowerCase() === 'excluded') {
    label = 'No';
  }
  else {
    label = value.toString();
  }
  return label;
};
