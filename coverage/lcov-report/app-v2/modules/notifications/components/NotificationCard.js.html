import React from 'react';
import PropTypes from "prop-types";
import { Link } from 'react-router';
 
import {
  GNotification,
  GNotificationData,
  GNotificationImg,
} from '../G';
import { timeFromNow } from 'helpers/DateTime';
import Avatar from 'modules/shared/Avatar';
 
const NotificationCard = ({ message }) => (
  <Link to={message.url}>
    <GNotification>{/* (message.read) ? isViewed : '' */}
      <GNotificationData className="bt">
        <GNotificationImg className="absolute l15">
          <Avatar imagePath={message.imageUrl} name={message.name || ''} />
        </GNotificationImg>
        <p className="pfc3 f14p m0">{message.text}</p>
        <p className="pfc4 f12 m0 mt4">{timeFromNow(message.time)}</p>
      </GNotificationData>
    </GNotification>
  </Link>
);
 
NotificationCard.propTypes = {
  message: PropTypes.object.isRequired,
};
 
export default NotificationCard;
