class CancelRequestPopUp extends React.Component {
 
  static propTypes = {
    requestedTrip: PropTypes.object.isRequired,
    handleSubmit: PropTypes.func.isRequired,
    location: PropTypes.object.isRequired,
    section: PropTypes.string.isRequired,
    closeModal: PropTypes.func.isRequired
  };
 
  static defaultProps = {};
 
  constructor() {
    super();
    this.state = {
      showTextClass: false,
      currentChecked: '',
      defaultOption: '',
      dropdowns: [false, false, false],
      textareas: [false, false, false]
    };
  }
 
  toggleDropdowns = (idx) => {
    const a = [false, false, false];
    a[idx] = true;
    this.setState({
      dropdowns: a
    });
    if (idx === 1) {
      this.setState({
        textareas: [false, true, false]
      });
    } else {
      this.setState({
        textareas: [false, false, false]
      });
    }
  };
 
  toggleTextareas = (idx, val) => {
    const a = [false, false, false];
 
    switch (idx) {
      case 0:
        if (val.value === notGoigOpt[1] || val.value === notGoigOpt[3])
          a[idx] = true;
        break;
      case 2:
        if (val.value === goingNotTTOpt[4]) {
          a[idx] = true;
        }
        break;
    }
 
    this.setState({
      textareas: a
    });
  };
 
  componentDidMount() {
    this.toggleDropdowns(0);
  }
 
  render() {
    const dropdownOne = this.state.dropdowns[0] ?
      (<Field
        name="reason0"
        options={0}
        component={RenderDropdown}
        onChange={(values) => this.toggleTextareas(0, values)}
      />) : null;
 
    const dropdownTwo = this.state.dropdowns[1] ?
      (<Field
        name="reason1"
        options={1}
        component={RenderDropdown}
      />) : null;
 
    const dropdownThree = this.state.dropdowns[2] ?
      (<Field
        name="reason2"
        options={2}
        component={RenderDropdown}
        onChange={(values) => this.toggleTextareas(2, values)}
      />) : null;
 
    const textareaOne = this.state.textareas[0] ?
      (<Field
        name="otherReason"
        component={RenderTextarea}
      />) : null;
 
    const textareaTwo = this.state.textareas[1] ?
      (<Field

      />) : null;

      />) : null;
 
    return (
      <div className="clearfix">

                <p className="f24 fw9 pfc1 mb0">Cancel Request</p>
                <p className="f16 fw9 pfc3 mb0">Are you sure you want to cancel your request for
                  trip to {this.props.requestedTrip.to_loc}? Please let us know why.</p>
              </div>
            </div>
            <div className="col-sm-12 p24 pl48 pr48 bb">
              <Field
                name="cancel_reason"
                component={RenderRadioButton}
                label={radioOptions[0]}
                id="not-going"
                type="radio"
                value={radioOptions[0]}
                onClickHandler={() => this.toggleDropdowns(0)}
              />
              {dropdownOne}
              {textareaOne}
              <Field
                name="cancel_reason"
                component={RenderRadioButton}
                label={radioOptions[1]}
                id="not-satisfied"
                type="radio"
                value={radioOptions[1]}
                onClickHandler={() => this.toggleDropdowns(1)}

              {dropdownTwo}

    );
  }
}
