import React from 'react';
import PropTypes from 'prop-types';
import { Link } from 'react-router';
 
import { getPageNameFromUrl } from 'helpers/FormDataFormatter';
 
import glamorous from 'glamorous';
 
const GBreadcrumOl = glamorous.ol({
  margin: '0',
  padding: '0',
  '& li': {
    display: 'inline-block',
    listStyle: 'none',
    marginRight: '3px',
    fontSize: '10px',
    lineHeight: '16px'
  },
  '& a': {
    color: '#2196F3',
  },
});
 
const Breadcrumb = ({ breadcrumbs }) => {
  const analyticsClick = (breadcrumb) => {
    const name = getPageNameFromUrl(breadcrumb.url);
    if (name === 'Package List Page') {
      const category = breadcrumb.url.split('/')[1].split('-')[0];
      const destination = breadcrumb.url.split('/')[2];
    }
  };
 
  return (
    <div className="clearfix">
      <GBreadcrumOl className="breadcrumb-list at_breadcrumb" itemScope itemType="https://schema.org/BreadcrumbList">
        {breadcrumbs.map((breadcrumb, index) =>
          <li
            key={`${breadcrumb.name}_${index}`}
            itemProp="itemListElement"
            itemScope
            itemType="https://schema.org/ListItem"
          >
            {
              (index !== breadcrumbs.length - 1) ?
                <Link
                  target="_blank"
                  to={breadcrumb.url}
                  itemProp="item"
                  onClick={() => analyticsClick(breadcrumb)}
                >
                  <span itemProp="name">{`${breadcrumb.name} > `}</span>
                </Link> :
                breadcrumb.name
            }
            <meta itemProp="position" content={index + 1} />
          </li>
        )}
      </GBreadcrumOl>
    </div>
  );
};
 
Breadcrumb.propTypes = {
  breadcrumbs: PropTypes.array.isRequired,
};
 
Breadcrumb.defaultProps = {
  breadcrumbs: [],
};
 
export default Breadcrumb;
